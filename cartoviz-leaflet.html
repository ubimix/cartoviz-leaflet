Examples

Afficher un fond de carte

<script type='text/javascript'>
var map = L.map('map').setView([ 48.86167, 2.33314], 13);
L.tileLayer('http://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: 'Map data © <a href="http://openstreetmap.org">OpenStreetMap</a>',
    maxZoom: 18
}).addTo(map);
</script>

Examples

Ajouter des marqueurs

<script type='text/javascript'>
 var marker = L.marker([ 48.86167, 2.33314]).addTo(map);
</script>

Examples

Ajouter un popup

<script type='text/javascript'>
marker.bindPopup("<b>Ici Simplon !</b>.").openPopup();
</script>

Examples

Clusteriser des données GeoJSON

<script type='text/javascript'>
var DATA = {"type": "FeatureCollection","features":[
    {"type": "Feature",
     "geometry": {"type": "Point","coordinates": ["2.3475","48.853611111111114"]},
     "properties": {"name": "Crypte archéologique du parvis Notre-Dame"
  }},...]};
var markers = new L.MarkerClusterGroup();
for (var i = 0; i < DATA.features.length; i++) {
    var coords = DATA.features[i].geometry.coordinates;
    var latlng = L.latlng(coords[1],coords[0]);
    var marker = new L.Marker(latlng);
    markers.addLayer(marker);
}
map.addLayer(markers);
</script>
